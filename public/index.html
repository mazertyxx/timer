let displayTime = 0; function format(t){ let m = Math.floor(t/60); let s = t%60; return `${String(m).padStart(2,'0')}:${String(s).padStart(2,'0')}`; } async function fetchState(){ const res = await fetch("/state"); const data = await res.json(); if(!data.running && data.time === 0){ document.getElementById("timer").style.display = "none"; return; } document.getElementById("timer").style.display = "block"; // animation si +30 if(data.time > displayTime){ document.getElementById("timer").style.color = "red"; document.getElementById("bonus").style.opacity = 1; setTimeout(()=>{ document.getElementById("timer").style.color = "white"; document.getElementById("bonus").style.opacity = 0; },1000); } displayTime = data.time; document.getElementById("timer").textContent = format(data.time); } // refresh toutes les 200ms setInterval(fetchState, 200);
